#include <Arduino.h>
#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "WT-MIGUEL";
const char* password = "GomesCampos9862709402";
const char* backendURL = "http://192.168.1.3:3000"; // URL do backend


void testarGet() {
  HTTPClient http;
  http.begin(String(backendURL) + "/test-get"); // Endpoint GET
  http.setTimeout(10000);
  int httpCode = http.GET();

  if (httpCode > 0) {
    String response = http.getString();
    Serial.println("Resposta do GET:");
    Serial.println(response);
  } else {
    Serial.printf("Erro no GET: %s\n", http.errorToString(httpCode).c_str());
  }

  http.end(); // Fecha a conexão
}

void testarPost() {
  HTTPClient http;
  http.begin(String(backendURL) + "/test-post"); // Endpoint POST
  http.addHeader("Content-Type", "application/json"); // Define o tipo de conteúdo
  http.setTimeout(10000);
  String payload = "{\"message\":\"Teste de POST\"}"; // Mensagem de teste
  int httpCode = http.POST(payload);

  if (httpCode > 0) {
    http.setTimeout(20000);
    String response = http.getString();
    Serial.println("Resposta do POST:");
    Serial.println(response);
  } else {
    Serial.printf("Erro no POST: %s\n", http.errorToString(httpCode).c_str());
  }

  http.end(); // Fecha a conexão
}

void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  Serial.print("Conectando ao WiFi...");
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.print(".");
  }
  Serial.println("\nConectado!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());

  // Testa a chamada GET
  testarGet();

  // Testa a chamada POST
  testarPost();
}
